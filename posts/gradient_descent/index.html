when we talk about gradient descent,often confuse: what is Gradient。

一维情况

to simipilify,we define cost function is$f(x)=x^2$

so,partial dilivitive of f is$\frac{\part f}{\part x}=2x$

$X:=X-\theta\frac{\part f}{\part x}$

import numpy as np
import matplotlib.pyplot as plt

#定义初始位置

cur_x=6
cur_y=(cur_x)**2

#learning rate

eta=0.1

iter=1000

#curent_gd

cur_df=None

#存储运行中的坐标

all_x=[]
all_y=[]

all_x.append(cur_x)
all_y.append(cur_y)

for i in range(iter):
    cur_df=2*cur_x

    cur_x=cur_x-eta*cur_df
    cur_y=(cur_x)**2

    all_x.append(cur_x)
    all_y.append(cur_y)

x=np.linspace(-1,7,100)
y=x**2

plt.plot(x,y)
plt.axis([-1, 10, 0, 40])

plt.scatter(np.array(all_x), np.array(all_y), color='red')
plt.show()
